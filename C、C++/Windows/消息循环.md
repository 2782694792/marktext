- Windows 应用程序中用来接收和处理用户输入、系统时间以及其他消息的一套机制；
- Win32 编程中，使用一个无限循环来不不断从消息队列中取出消息并进行处理；
- WndProc 消息函数在收到消息后发送退出消息，以退出消息循环；

```cpp
#include <windows.h>

LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam);

int WINAPI WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance, LPSTR lpCmdLine, int nCmdShow)
{
    // 1. 注册窗口类
    WNDCLASS wc = {};
    wc.lpfnWndProc = WndProc; // 指定窗口过程函数
    wc.hInstance = hInstance;
    wc.lpszClassName = L"MessageLoopExample";
    RegisterClass(&wc);

    // 2. 创建窗口
    HWND hWnd = CreateWindow(wc.lpszClassName, L"Message Loop Example", WS_OVERLAPPEDWINDOW,
        CW_USEDEFAULT, CW_USEDEFAULT, 400, 300, NULL, NULL, hInstance, NULL);

    if (!hWnd) {
        return -1;
    }

    // 3. 显示窗口
    ShowWindow(hWnd, nCmdShow);
    UpdateWindow(hWnd);

    // 4. 消息循环
    MSG msg = {};
    while (GetMessage(&msg, NULL, 0, 0)) { // 从消息队列中取出消息
        TranslateMessage(&msg); // 转换键盘消息
        DispatchMessage(&msg); // 分派消息给窗口过程函数
    }

    return msg.wParam;
}

LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
{
    switch (message)
    {
    case WM_DESTROY:
        PostQuitMessage(0); // 当窗口关闭时发送退出消息
        break;
    default:
        return DefWindowProc(hWnd, message, wParam, lParam);
    }
    return 0;
}
```
